dataset = await redivis
  .user('stanford_templates')
  .dataset('item_response_warehouse')

tables = await dataset.listTables()
dataset_names = await tables.map(t => t.name).filter(s => !s.startsWith('_')).sort()
datasets = new Map()

async function getTable(tableName) {
  const rows = await redivis
    .user('stanford_templates')
    .dataset('item_response_warehouse')
    .table(tableName)
    .listRows({ maxResults: 10 })
  return rows
}

async function getData(name) {
  if (! datasets.has(name)) {
    const table = await getTable(name);
    datasets.set(name, table);
  }
  return datasets.get(name)
}

Programmatic access

You can also access IRW data programmatically using the Redivis API R or Python. For example:

  • R
  • Python
# individual dataset
dataset <- redivis::user("stanford_templates")$dataset("item_response_warehouse")
df <- dataset$table("16_personalityfactors")$to_tibble()

# metadata
project <- redivis::user("mikabr")$project("irw")
metadata <- project$table("metadata_output")$to_tibble()
# individual dataset
dataset = redivis.user('stanford_templates').dataset('item_response_warehouse')
df = dataset.table('16_personalityfactors').to_pandas_dataframe()

# metadata
project = redivis.user('mikabr').project('irw')
metadata = project.table('metadata_output').to_pandas_dataframe()
